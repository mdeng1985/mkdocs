    WSL下的实践¶

    准备知识¶

    1. WSL下开启SSH登录，参考TAGS_LINUX_WSL2_INSTALL
    2. WSL创建cmake_ws, 参考TAGS_CMAKE_PRACTICE
    3. VS2019下连接远程WSL，并配置好WSL开发, 打开cmake_ws目录下的CMakelists.txt 参考TAGS_CMAKE_PRACTICE，
    4. WSL下安装eigen3: sudo apt-get install libeigen3-dev

    编写¶

    创建源文件eigenMatrix.cpp和CMakeLists.txt （参考的是视觉SLAM十四讲的第3章）,两者均位于cmake_ws/src/slambook2/ch3/src目录下，其中cpp内容如下： 

        #include <iostream>
using namespace std;

#include <ctime>
#include <Eigen/Core>
#include <Eigen/Dense>
using namespace Eigen;

#define MATRIX_SIZE 50

/****************************
* 本程序演示了 Eigen 基本类型的使用
****************************/

int main(int argc,char **argv){
    Matrix<float,2,3> matrix_23;//2x3的float型矩阵
    matrix_23<<1,2,3,4,5,6;
     // 用()访问矩阵中的元素
    cout << "print matrix 2x3: " << endl;
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) cout << matrix_23(i, j) << "\t";
        cout << endl;
    }

    cout<<"matrix 2x3 from 1 to 6:\n" << matrix_23 <<endl;

    //3d矩阵
    Vector3d v_3d;//double
    Matrix<float,3,1> vd_3d;//float

    Matrix3d matrix_33 = Matrix3d::Zero();

    Matrix<double,Dynamic,Dynamic> matrix_dynamic;
    MatrixXd matirx_x;


    //矩阵相乘
    v_3d << 3,2,1;
    vd_3d <<4,5,6;

    Matrix<double,2,1> result = matrix_23.cast<double>() * v_3d;

    cout << "[1,2,3;4,5,6]*[3,2,1]=" << result.transpose() << endl;

    Matrix<float, 2, 1> result2 = matrix_23 * vd_3d;//float

    cout << "[1,2,3;4,5,6]*[3,2,1]=" << result2.transpose() << endl;


    //四则运算
    matrix_33 = Matrix3d::Random();
    cout << "random matrix:\n" << matrix_33 << endl;
    cout << "transpose:\n" << matrix_33.transpose() << endl;
    cout << "sum:" << matrix_33.sum() << endl;
    cout << "trace:" << matrix_33.trace() << endl;
    cout << "times 10: \n" << 10 * matrix_33 << endl;
    cout << "inverse: \n" << matrix_33.inverse() << endl;        // 逆
    cout << "det: " << matrix_33.determinant() << endl;    // 行列式


    //特征值
    SelfAdjointEigenSolver<Matrix3d> eigen_solver(matrix_33.transpose() * matrix_33);
    cout << "Eigen value =\n" << eigen_solver.eigenvalues() << endl;
    cout << "Eigen vectors =\n" << eigen_solver.eigenvectors() << endl;


    //解方程

    Matrix<double, MATRIX_SIZE,MATRIX_SIZE> matrix_NN= MatrixXd::Random(MATRIX_SIZE, MATRIX_SIZE);
    matrix_NN = matrix_NN * matrix_NN.transpose();

    Matrix<double, MATRIX_SIZE, 1> v_Nd = MatrixXd::Random(MATRIX_SIZE, 1);
    //直接求逆

    clock_t time_stt = clock(); // 计时
    Matrix<double, MATRIX_SIZE, 1> x = matrix_NN.inverse() * v_Nd;

    cout << "time of normal inverse is "
        << 1000 * (clock() - time_stt) / (double)CLOCKS_PER_SEC << "ms" << endl;
    cout << "x = " << x.transpose() << endl;

    //QR分解 求解
    time_stt = clock();
    x = matrix_NN.colPivHouseholderQr().solve(v_Nd);
    cout << "time of Qr decomposition is "
        << 1000 * (clock() - time_stt) / (double)CLOCKS_PER_SEC << "ms" << endl;
    cout << "x = " << x.transpose() << endl;
    // 对于正定矩阵，还可以用cholesky分解来解方程
    time_stt = clock();
    x = matrix_NN.ldlt().solve(v_Nd);

    cout << "time of ldlt decomposition is "
        << 1000 * (clock() - time_stt) / (double)CLOCKS_PER_SEC << "ms" << endl;
    cout << "x = " << x.transpose() << endl;

}

    CMakeLists.txt 内容如下： 

        cmake_minimum_required(VERSION 3.5.1)
    project(useEigen)
    SET(CMAKE_BUILD_TYPE "Debug")
    SET(CMAKE_CXX_FLAGS_DEBUG "$ENV{CXXFLAGS} -O0 -Wall -g2 -ggdb")
    # SET(CMAKE_CXX_FLAGS_DEBUG "$ENV{CXXFLAGS} -Wall   -ggdb3 -O0   -gcodeview -std=c99")
    SET(CMAKE_CXX_FLAGS_RELEASE "$ENV{CXXFLAGS} -O3 -Wall")

    include_directories("/usr/include/eigen3")
    SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/../../../../bin) 
    SET(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/../../../../libs)
    add_executable(useEigen eigenMatrix.cpp)

    注意：SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/../../../../bin)只是在wsl自身下编译才起作用，对于vs2019远程编译好像不会在这个目录下生成bin文件

    vs2019远程开发¶

    project(cmake_ws)
    add_subdirectory("src/slambook2/ch3/src")

    vscode wsl下运行¶

    在终端的cmake_ws目录下 

    cd build
cmake ../src/slambook2/ch3/src/
make 
../bin/useEigen

    windows系统下的实践¶

    1. 下载eigen3的源文件，解压缩（D:\WORK\MyCode\cppLib\source\eigen3_3）

    2. 打开cmake gui源文件设置为D:\mdeng\WORK\MyCode\cppLib\source\eigen3_3，where to build ..设置为D:\mdeng\WORK\MyCode\cppLib\source\eigen3_3\build

    3. Configure以后： CMAKE_INSTALL_PREFIX为eigen3的编译后的目录 默认为：C:/Program Files (x86)/Eigen3，可以自己设置D:/mdeng/WORK/MyCode/cppLib/Eigen3

    4. Generate

    5. 然后以管理员身份打开vs，在vs里打开build下的eigen3.sln，然后右键点All build项目，生成；生成完以后点install项目，右键生成(在默认的方案解决平台win32下生成的)。生成完以后，编译好的eigen3文件就出现在D:/WORK/MyCode/cppLib/Eigen3了

    6. 测试，新建一个项目，然后添加包含目录D:\WORK\MyCode\cppLib\Eigen3\include\eigen3就可以使用了。程序见testEigen。

    备注：对于有些标准的库和包含文件，需要设置成所有项目都包含。添加方式如下： view --property manager-- debug| Win32- microsoft.CPP.win32.user 同样的，在x64平台也需要添加view --property manager-- debug| x64- microsoft.CPP.win64.user。这样就不需要每个项目都需要设置了。 参考complier_study.rmd

    其他注意事项：

    • 由于eigen3没有lib，只用在debug模式下生成即可。
    • eigen也没有库文件，也不用在多个平台上进行编译
