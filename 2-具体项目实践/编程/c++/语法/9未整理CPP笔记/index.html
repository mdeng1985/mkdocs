    iomanip¶

    iomanip头文件是用于操作输入输出的。

    假设源文件testIomanip.cpp为

    /* Test Formatting Output (TestFormattedOutput.cpp) */
#include <iostream>
#include <iomanip>    // Needed to do formatted I/O
using namespace std;

int main() {
   // Floating point numbers
   double pi = 3.14159265;
   // 下面这一行是
   cout << fixed << setprecision(4); // fixed format with 4 decimal places
   cout << pi << endl;
   cout << "|" << setw(8) << pi << "|" << setw(10) << pi << "|" << endl;
      // setw() is not sticky, only apply to the next operation.
   cout << setfill('-');
   cout << "|" << setw(8) << pi << "|" << setw(10) << pi << "|" << endl;
   cout << scientific;  // in scientific format with exponent
   cout << pi << endl;

   // booleans
   bool done = false;
   cout << done << endl;  // print 0 (for false) or 1 (for true)
   cout << boolalpha;     // print true or false
   cout << done << endl;
   return 0;
}

    编译以后运行为：

    3.1416
|  3.1416|    3.1416|
|--3.1416|----3.1416|
3.1416e+000
0
false
    具体解释

    • setw(int field-widht): set the field width for the next IO operation. setw() is non-sticky and must be issued prior to each IO operation. The field width is reset to the default after each operation (with just enough width to accommodate the field).

    • setfill(char fill-char): set the filled character for padding to the field width.

    • left|right|internal: set the alignment

    • fixed/scientific (for floating-point numbers): use fixed-point notation (e.g, 12.34) or scientific notation (e.g., 1.23e+006).

    • setprecision(int numDecimalDigits) (for floating-point numbers): specify the number of digits after the decimal point.

    • boolalpha/noboolalpha (for bool): display bool values as alphabetic string (true/false) or 1/0.

    将数组传递到函数里，不仅要传递数组本身，还需要传递数组的size。参考下例。

        /* Function to compute the sum of an array (SumArray.cpp) */
    #include <iostream>
    using namespace std;

    // Function prototype
    int sum(int array[], int size);    // Need to pass the array size too
    void print(int array[], int size);

    // Test Driver
    int main() {
       int a1[] = {8, 4, 5, 3, 2};
       print(a1, 5);   // {8,4,5,3,2}
       cout << "sum is " << sum(a1, 5) << endl;  // sum is 22
    }

    // Function definition
    // Return the sum of the given array
    int sum(int array[], int size) {
       int sum = 0;
       for (int i = 0; i < size; ++i) {
          sum += array[i];
       }
       return sum;
    }

    // Print the contents of the given array
    void print(int array[], int size) {
       cout << "{";
       for (int i = 0; i < size; ++i) {
          cout << array[i];
          if (i < size - 1) {
             cout << ",";
          }
       }
       cout << "}" << endl;
    }

    传递值VS传递reference¶

    • 值传递：将变量克隆一个，传递这个克隆到函数里，因此函数不管做什么对原来的变量没有影响。在C/C++中，基本类型是以值传递的。

    • 传递reference（相当于传递别名）： 函数里的值变了会影响外面的值。数组传递就是reference传递。参考下例

        /* Function to increment each element of an array (IncrementArray.cpp) */
    #include <iostream>
    using namespace std;

    // Function prototypes
    void inc(int array[], int size);
    void print(int array[], int size);

    // Test Driver
    int main() {
       int a1[] = {8, 4, 5, 3, 2};

       // Before increment
       print(a1, 5);   // {8,4,5,3,2}
       // Do increment
       inc(a1, 5);     // Array is passed by reference (having side effect)
       // After increment
       print(a1, 5);   // {9,5,6,4,3}
    }

    // Function definitions

    // Increment each element of the given array
    void inc(int array[], int size) {  // array[] is not const
       for (int i = 0; i < size; ++i) {
          array[i]++;  // side-effect
       }
    }

    // Print the contents of the given array
    void print(int array[], int size) {
       cout << "{";
       for (int i = 0; i < size; ++i) {
          cout << array[i];
          if (i < size - 1) {
             cout << ",";
          }
       }
       cout << "}" << endl;
    }
    • 如果传递数组到函数里，不像数组的值被函数改变，可以加上const关键词。
        void squareA(const double a1[], double a2[], int size)
    {
    for (int i=0;i<size;i++)
    {
       // a1[i] = a1[i]+1; //a1的值不允许在函数里改变
        a2[i] = a2[i]+1;
    }
    }
    • 如果想以reference的形式传递基本类型，可以在前面加上 &。
    /* Test Pass-by-reference for fundamental-type parameter
   via reference declaration (TestPassByReference.cpp) */
#include <iostream>
using namespace std;

int squareByValue (int number);        // Pass-by-value
void squareByReference (int & number); // Pass-by-reference

int main() {
   int n1 = 8;
   cout << "Before call, value is " << n1 << endl;  // 8
   cout << squareByValue(n1) << endl;  // no side-effect
   cout << "After call, value is " << n1 << endl;   // 8

   int n2 = 9;
   cout << "Before call, value is " << n2 << endl;  // 9
   squareByReference(n2);  // side-effect
   cout << "After call, value is " << n2 << endl;   // 81
}

// Pass parameter by value - no side effect
int squareByValue (int number) {
   return number * number;
}

// Pass parameter by reference by declaring as reference (&)
// - with side effect to the caller
void squareByReference (int & number) {
   number = number * number;
}

    class, instance 和 object¶

    • class 抽象的类，相当于模板，好比是人的统称

    • instance 具体实现，比如说张三

    • object 通常指的是instance，但有时候也泛指一般的class

    class的三部分¶

    • 类名： 首字母大写，比如说 ScoeerPlayer

    • 数据变量：数据，用名字，首字母小，比如radius

    • 函数：动态，用动词，首字母小写，比如getRadius

    数据变量和函数统称为member（成员）

    法则

    • 数据变量的属性通常设置为prviate，这意味着仅能在内部调用，不能直接从外部调用: r1.radius是不对的

    • 函数可以设置为pubilc，允许外部调用。

    • 调用类里面成员的操作符是.，比如说r1.getRadius()

    关于类中的函数¶

    函数声明和定义¶

    参考资料: c++ primer的7.1.3节

    类中的函数包括**成员函数**和**非成员函数**， 这两种函数的声明都是在头文件中。

    • 成员函数的声明在类的大括号的内部，相应的调用格式为 类名.函数名
    • 非成员函数的声明在类的大括号的外部，相应的调用格式直接用函数名，不要在前面加类名(有点像面向过程)

    • data.combine()
    • add(...)

    关于构造器¶

    构造器的几点说明¶

    • 构造器是特殊的函数，它的名字和类名一致

    • 构造器没有返回类型，也就是没有return

    • 构造器只能调用一次，也就是在对instance初始化的时候调用

    • 构造器不能继承

                     // Take note that there is no empty bracket ()
Circle c1 = Circle(1.2, "red");  // radius, color
Circle c2 = Circle(3.4);         // radius, default color
Circle c3 = Circle();            // default radius and color

    Circle c3(); //系统会以为是调用c3函数

    this,const，::和.¶

    this相当于一个指针，指向类内部的数据变量, 比如

    class Circle {
private:
   double radius;                 // Member variable called "radius"
   ......
public:
   void setRadius(double radius) { // Function's argument also called "radius"
      this->radius = radius;
         // "this.radius" refers to this instance's member variable
         // "radius" resolved to the function's argument.
   }
   ......
}

    const 用在动态函数后面，说明函数内部不允许修改类的数据变量，比如

    double getRadius() const {  // const member function
   radius = 0;  
      // error: assignment of data-member 'Circle::radius' in read-only structure
   return radius;
}

    ::为class scope resolution operator，在类的执行文件(.cpp)里定义每个动态函数的时候前面需要加上类名::函数名， 比如Circle.cpp：

   radius = r;
   color = c;
}

// Public getter for private data member radius
double Circle::getRadius() const {
   return radius;
}
}

    头文件和执行文件¶

    在编程的时候，一般将程序分成头文件(.h，include file)和执行文件(.cpp)两部分。头文件中给出声明，执行文件中给出定义。

    头文件Circle.h

        /* The Circle class Header (Circle.h) */
    #include <string>   // using string
    using namespace std;

    // Circle class declaration
    class Circle {
    private:   // Accessible by members of this class only
       // private data members (variables)
       double radius;
       string color;

    public:    // Accessible by ALL
       // Declare prototype of member functions
       // Constructor with default values
       Circle(double radius = 1.0, string color = "red");

       // Public getters & setters for private data members
       double getRadius() const;
       void setRadius(double radius);
       string getColor() const;
       void setColor(string color);

       // Public member Function
       double getArea() const;
    };

    Circle.cpp

        /* The Circle class Implementation (Circle.cpp) */
    #include "Circle.h" // user-defined header in the same directory

    // Constructor
    // default values shall only be specified in the declaration,
    // cannot be repeated in definition
    Circle::Circle(double r, string c) {
       radius = r;
       color = c;
    }

    // Public getter for private data member radius
    double Circle::getRadius() const {
       return radius;
    }

    // Public setter for private data member radius
    void Circle::setRadius(double r) {
       radius = r;
    }

    // Public getter for private data member color
    string Circle::getColor() const {
       return color;
    }

    // Public setter for private data member color
    void Circle::setColor(string c) {
       color = c;
    }

    // A public member function
    double Circle::getArea() const {
       return radius*radius*3.14159265;
    }

    几点说明：

    1. 数据变量中的声明是在头文件中给出的，在执行文件中不用再做声明

    2. 执行文件需要include头文件：#include "Circle.h"，如果为双引号，则编译器会在当前目录下找，如果为<>，则编译器仅会在系统目录下找头文件(不会在cpp所在目录下找)

    3. 执行文件的所有函数的定义前面要加上Circle::

    4. 构造器的数据变量的默认参数是在头文件中给出，不能在执行文件中给出，Circle(double radius = 1.0, string color = "red")

    5. 执行文件中，构造器的格式还可以是Circle::Circle(double r, string c) : radius(r)，color(c) {}。括号里是传递的参数名，括号以外是类的数据变量名。

    指针¶

    指针存储的是变量内存地址。

    关于*¶

    • 用于声明语句： int *iPtr， 表明iPtr为指针，存储的是整型变量的地址。在声明的时候从右边解释：iPtr为指针，该指针指向的变量为整型(该指针存放的变量为整型变量的地址)

    • 用于Dereferencing操作，比如指针iPtr指向变量a，则*iPtr得到的是a的值。

    • 怎么判断是声明还是dereference(解引用)： 如果*前面为一个类型（比如int），则为声明，否则为解引用。

    • 如果两个指针存放的地址是一样的，则它们相等。而在如下三种情况下，可以判定为两个指针的存放地址是一样的: 1. 存放的是同一个对象的地址; 2. 均为空指针; 3. they are both pointers one past the same object. (c++primer 的2.3节)

    比如:

    int number = 88;
int * pNumber = &number;  // Declare and assign the address of variable number to pointer pNumber (0x22ccec)
cout << pNumber<< endl;   // Print the content of the pointer variable, which contain an address (0x22ccec)
cout << *pNumber << endl; // Print the value "pointed to" by the pointer, which is an int (88)
*pNumber = 99;            // Assign a value to where the pointer is pointed to, NOT to the pointer variable
cout << *pNumber << endl; // Print the new value "pointed to" by the pointer (99)
cout << number << endl;   // The value of variable number changes as well (99)
    Dereferencing操作可以既可以用于等式左边，又可以用于右边，比如

    *sPtr = 3;
int temp = *sPtr

    指针初始化¶

    方式1 

    int number = 88;     // An int variable with a value
int * pNumber;       // Declare a pointer variable called pNumber pointing to an int (or int pointer)
pNumber = &number;   // Assign the address of the variable number to pointer pNumber

    方式2 

    int * pAnother = &number; (定义指针pAnother，其值为变量number的地址)
    方式3:空指针

    int * p = NULL;

    方式4：动态分配内存(new) 

    int * p2;            // Not initialize, points to somewhere which is invalid
cout << p2 << endl; // Print address before allocation
p2 = new int;       // Dynamically allocate an int and assign its address to pointer
*p2 = 3;
    用完以后再delete p2。但是下面的用法是不对的(没有动态分配内存，不知道指针指向哪)

    int * iPtr;
*iPtr = 55;
cout << *iPtr << endl;

    注意：

    • 如果指针类型为数组，则对应的是new[]和delete[]

    比如：

    int * pArray;
pArray = new int[5];
....
delete[] pArray;
    • 指针可以指向不同的变量，但是指向的类型不能改变。

    reference 变量¶

    reference变量相当于别名，几点注意事项：

    • 声明：采用&，比如int &a=b 表明a是b的别名。a和b其中一个的值改变了，另外一个也相应改变。技巧：在声明的时候从右边解释比较好，比如int &a表明a为一个别名，且这个别名所对应的对象为整形变量。

    • &如果用在等式右边表明取地址

    - reference变量必须初始化,而且一旦建立别名了，不能将其改变成另外一个变量的别名。¶

    • refrence必须是object的别名，不是具体数值，比如int &a=10;是不正确的
      • 怎么判断是声明还是取地址： 如果*前面为一个类型（比如int），则为声明，否则为取地址。

    再探值传递和reference传递¶

    值传递相当于clone。主要阐述下reference传递。

    首先看例子1：

    /* Pass-by-reference using pointer (TestPassByPointer.cpp) */
#include <iostream>
using namespace std;

void square(int *);

int main() {
   int number = 8;
   cout <<  "In main(): " << &number << endl;  // 0x22ff1c
   cout << number << endl;   // 8
   square(&number);          // Explicit referencing to pass an address
   cout << number << endl;   // 64
}

void square(int * pNumber) {  // Function takes an int pointer (non-const)
   cout <<  "In square(): " << pNumber << endl;  // 0x22ff1c
   *pNumber *= *pNumber;      // Explicit de-referencing to get the value pointed-to
}
    - void square(int * pNumber)函数的输入变量为指针，由于指针的内容为地址，因此调用的时候采取&number

    • *pNumber *= *pNumber 也即*pNumber = *pNumber * *pNumber *pNumber指向的变量的平方。（地址没变）

    例子2

    /* Pass-by-reference using reference (TestPassByReference.cpp) */
#include <iostream>
using namespace std;

void square(int &);

int main() {
   int number = 8;
   cout <<  "In main(): " << &number << endl;  // 0x22ff1c
   cout << number << endl;  // 8
   square(number);          // Implicit referencing (without '&')
   cout << number << endl;  // 64
}

void square(int & rNumber) {  // Function takes an int reference (non-const)
   cout <<  "In square(): " << &rNumber << endl;  // 0x22ff1c
   rNumber *= rNumber;        // Implicit de-referencing (without '*')
}

    • void square(int & rNumber) 输入变量的类型为int &为别名，因此number是rNumber的别名

    • 调用的时候就不需要加取地址运算符。

    Passing the Function's Return Value¶

    /* Passing back return value using reference (TestPassByReferenceReturn.cpp) */
#include <iostream>
using namespace std;

int & squareRef(int &); //说明函数返回的是别名
int * squarePtr(int *); //说明函数返回的是指针

int main() {
   int number1 = 8;
   cout <<  "In main() &number1: " << &number1 << endl;  // 0x22ff14
   int & result = squareRef(number1);
   cout <<  "In main() &result: " << &result << endl;  // 0x22ff14
   cout << result << endl;   // 64
   cout << number1 << endl;  // 64

   int number2 = 9;
   cout <<  "In main() &number2: " << &number2 << endl;  // 0x22ff10
   int * pResult = squarePtr(&number2);
   cout <<  "In main() pResult: " << pResult << endl;  // 0x22ff10
   cout << *pResult << endl;   // 81
   cout << number2 << endl;    // 81
}

int & squareRef(int & rNumber) {
   cout <<  "In squareRef(): " << &rNumber << endl;  // 0x22ff14
   rNumber *= rNumber;
   return rNumber;
}

int * squarePtr(int * pNumber) {
   cout <<  "In squarePtr(): " << pNumber << endl;  // 0x22ff10
   *pNumber *= *pNumber;
   return pNumber;
}

    指针和别名的多重操作¶

    • int **ppi = &pi 从右边开始读： ppi为指针，该指针存放的是另外一个指针pi的地址，也就是该ppi指针指向的是另外一个指针pi
    • int *&ppi= p 从右边开始读:ppi为指针p的别名。
